	bool tooTall = 0;
	int length = 0;
	int bikes = 0;
	double vehicleCost = 0;
	double cost = 0;

	cout << "Welcome to Matthew’s Fare Calculator";

	cout << "Are you driving a vehicle onto the ferry ? (y / n) : ";
	cin >> charToBool;
	if (charToBool == "y") {
		vehicle = true;
	} else if (charToBool == "n") {
		vehicle = false;
	}

	if (vehicle) {
		cout << "Is the driver a senior citizen (age 65+) or disabled ? (y / n) : ";
		cin >> charToBool;
		if (charToBool == "y") {
			driverStatus = true;
		} else if (charToBool == "n") {
			driverStatus = false;
		}

		cout << "How many passengers in your vehicle ? (excluding the driver)";
	}
	
	cout << "Adults (age 19 – 64) : ";
	cin >> adults;
	
	cout << "Senior Citizens (65 or older), or disabled Persons : ";
	cin >> seniors;
	
	cout << "Youth (age 5 - 18) : ";
	cin >> youths;

	if (vehicle) {
		cout << "Is your vehicle over 7 feet, 6 inches in height ? (y / n) : ";
		cin >> charToBool;
		if (charToBool == "y") {
			tooTall = true;
		}
		else if (charToBool == "n") {
			tooTall = false;
		}

		cout << "How long is your vehicle in feet ? ";
		cin >> length;
	}
	
	cout << "How many people in your group are traveling with a bicycle ? ";
	cin >> bikes;
	
	if (length == 0) {
		vehicleCost = 0;
	}
	else if (length < 14) {
		if (tooTall) {
			if (driverStatus) {
				vehicleCost = 65.35;
			} else {
				vehicleCost = 35.80;
			}
		} else {
			if (driverStatus) {
				vehicleCost = 29.55;
			} else {
				vehicleCost = 36.05;
			}
		}
	} else if (length < 22) {
		if (tooTall) {
			if (driverStatus) {
				vehicleCost = 83.25;
			} else {
				vehicleCost = 44.75;
			}
		} else {

				vehicleCost = 45.00;

	} else if (length < 50) {
